    <script>

        // Arrays para armazenar as transações

        let transacoes = [];

        // Variável para armazenar o total da poupança

        let totalPoupanca = 0;

        // Referências aos elementos do DOM

        const tipoTransacaoSelect = document.getElementById('tipo-transacao');

        const categoriaTransacaoSelect = document.getElementById('categoria-transacao');

        const descricaoTransacaoInput = document.getElementById('descricao-transacao');

        const valorTransacaoInput = document.getElementById('valor-transacao');

        const btnAdicionarTransacao = document.getElementById('btn-adicionar-transacao');

        const listaTransacoesDiv = document.getElementById('lista-transacoes');

        const totalReceitasSpan = document.getElementById('total-receitas');

        const totalDespesasSpan = document.getElementById('total-despesas');

        const saldoAtualSpan = document.getElementById('saldo-atual');

        const valorPoupancaInput = document.getElementById('valor-poupanca');

        const btnGuardarPoupanca = document.getElementById('btn-guardar-poupanca');

        const totalPoupancaSpan = document.getElementById('total-poupanca');

        // Categorias de receitas e despesas para jovens

        const categorias = {

            receita: ['Mesada', 'Presentes', 'Trabalhos Eventuais (Bico)', 'Ajuda da Família', 'Outras Receitas'],

            despesa: ['Lanche/Comida', 'Transporte', 'Lazer/Diversão', 'Estudos', 'Roupas/Estilo', 'Tecnologia', 'Presentes', 'Outras Despesas']

        };

        // Função para preencher as categorias no select

        function preencherCategorias() {

            const tipoSelecionado = tipoTransacaoSelect.value;

            const categoriasDisponiveis = categorias[tipoSelecionado];

            categoriaTransacaoSelect.innerHTML = ''; // Limpa as opções atuais

            categoriasDisponiveis.forEach(categoria => {

                const option = document.createElement('option');

                option.value = categoria;

                option.textContent = categoria;

                categoriaTransacaoSelect.appendChild(option);

            });

        }

        // Função para adicionar uma nova transação

        function adicionarTransacao() {

            const tipo = tipoTransacaoSelect.value;

            const categoria = categoriaTransacaoSelect.value;

            const descricao = descricaoTransacaoInput.value.trim(); // Remove espaços em branco

            const valor = parseFloat(valorTransacaoInput.value); // Converte para número

            // Validação básica dos campos

            if (!descricao || isNaN(valor) || valor <= 0) {

                alert('Por favor, preencha a descrição e o valor corretamente.');

                return;

            }

            // Cria um objeto para a transação

            const novaTransacao = {

                id: Date.now(), // ID único baseado no timestamp

                tipo: tipo,

                categoria: categoria,

                descricao: descricao,

                valor: valor

            };

            // Adiciona a nova transação ao array

            transacoes.push(novaTransacao);

            // Atualiza a exibição das transações e os totais

            atualizarListaTransacoes();

            atualizarResumoFinanceiro();

            // Limpa os campos do formulário

            descricaoTransacaoInput.value = '';

            valorTransacaoInput.value = '';

        }

        // Função para remover uma transação

        function removerTransacao(id) {

            // Filtra o array, removendo a transação com o ID correspondente

            transacoes = transacoes.filter(transacao => transacao.id !== id);

            // Atualiza a exibição e os totais

            atualizarListaTransacoes();

            atualizarResumoFinanceiro();

        }

        // Função para atualizar a lista de transações exibida

        function atualizarListaTransacoes() {

            listaTransacoesDiv.innerHTML = ''; // Limpa a lista atual

            if (transacoes.length === 0) {

                listaTransacoesDiv.innerHTML = '<p class="text-center text-gray-500">Nenhuma transação registrada ainda.</p>';

                return;

            }

            // Adiciona cada transação à lista

            transacoes.forEach(transacao => {

                const divTransacao = document.createElement('div');

                divTransacao.classList.add('transacao-item');

                const valorFormatado = `R$ ${transacao.valor.toFixed(2).replace('.', ',')}`; // Formata o valor

                // Define a classe CSS para cor do valor

                const classeValor = transacao.tipo === 'receita' ? 'valor-receita' : 'valor-despesa';

                divTransacao.innerHTML = `

                    <div>

                        <p class="text-gray-800 font-semibold">${transacao.descricao}</p>

                        <p class="text-gray-500 text-sm">${transacao.categoria}</p>

                    </div>

                    <div class="flex items-center gap-2">

                        <p class="${classeValor}">${valorFormatado}</p>

                        <button class="btn-remover" onclick="removerTransacao(${transacao.id})">Remover</button>

                    </div>

                `;

                listaTransacoesDiv.appendChild(divTransacao);

            });

        }

        // Função para atualizar o resumo financeiro (totais e saldo)

        function atualizarResumoFinanceiro() {

            let totalReceitas = 0;

            let totalDespesas = 0;

            // Calcula os totais

            transacoes.forEach(transacao => {

                if (transacao.tipo === 'receita') {

                    totalReceitas += transacao.valor;

                } else {

                    totalDespesas += transacao.valor;

                }

            });

            const saldo = totalReceitas - totalDespesas;

            // Atualiza os elementos no DOM com os valores formatados

            totalReceitasSpan.textContent = `R$ ${totalReceitas.toFixed(2).replace('.', ',')}`;

            totalDespesasSpan.textContent = `R$ ${totalDespesas.toFixed(2).replace('.', ',')}`;

            saldoAtualSpan.textContent = `R$ ${saldo.toFixed(2).replace('.', ',')}`;

            // Altera a cor do saldo com base no valor

            if (saldo >= 0) {

                saldoAtualSpan.classList.remove('text-red-600');

                saldoAtualSpan.classList.add('text-green-600');

            } else {

                 saldoAtualSpan.classList.remove('text-green-600');

                 saldoAtualSpan.classList.add('text-red-600');

            }

        }

        // Função para guardar dinheiro na poupança

        function guardarPoupanca() {

            const valor = parseFloat(valorPoupancaInput.value);

            if (isNaN(valor) || valor <= 0) {

                 alert('Por favor, insira um valor válido para guardar.');

                 return;

            }

            totalPoupanca += valor; // Adiciona o valor ao total da poupança

            totalPoupancaSpan.textContent = `R$ ${totalPoupanca.toFixed(2).replace('.', ',')}`; // Atualiza a exibição

            valorPoupancaInput.value = ''; // Limpa o campo de input

             alert(`R$ ${valor.toFixed(2).replace('.', ',')} guardados na poupança!`); // Feedback para o usuário

        }

        // Event Listeners

        // Atualiza as categorias quando o tipo de transação muda

        tipoTransacaoSelect.addEventListener('change', preencherCategorias);

        // Adiciona transação ao clicar no botão

        btnAdicionarTransacao.addEventListener('click', adicionarTransacao);

        // Adiciona valor à poupança ao clicar no botão

        btnGuardarPoupanca.addEventListener('click', guardarPoupanca);

        // Inicialização

        preencherCategorias(); // Preenche as categorias ao carregar a página

        atualizarResumoFinanceiro(); // Atualiza o resumo inicial (será R$ 0.00)

        // Nota: A funcionalidade de gráficos é mais complexa e exigiria uma biblioteca (como Chart.js)

        // ou implementação manual com Canvas/SVG, o que está além do escopo desta versão inicial simplificada.

        // A seção de gráficos foi incluída como placeholder.

    </script>
